1.	Сосчитать количество клиентов, покупающих товар.

SELECT COUNT(Client) AS [Количество клиентов]
FROM Client

2.	Сколько посредников поставляют товар для фирмы.

SELECT COUNT(Seller) AS [Количество посредников]
FROM Seller
WHERE Status LIKE 'посредник';

3.	Вывести среднюю сумму сделки по проданному товару.

SELECT AVG(CostUnit*Count) AS [СредняяСумма товара]
FROM ClientGoods

4.	Посчитать на какую сумму было всего поставлено товара.

SELECT SUM(CostUnit*Count) AS [Сумма товара]
FROM ClientGoods

5.	Определите максимальное количество закупленного товара шт.(кг)

SELECT Max(Count) AS [Максимальное кол-во товара]
FROM SallerGoods

6.	Какая минимальная сумма покупки, на которую клиент сделал заказ.

SELECT Min(CostUnit*Count) AS [минимальная сумма покупки]
FROM ClientGoods

7.	Посчитать на какую сумму было продано товара магазинам.

SELECT SUM(CostUnit * Count) AS 'сумма'
FROM ClientGoods cg, Client c
WHERE cg.INNClient=c.INNClient AND Status LIKE 'магазин'

8.	Сосчитать общее количество видов закупленных фирмой вафель.

SELECT COUNT(DISTINCT Goods) AS 'общее количество видов закупленных фирмой вафель'
FROM SallerGoods
WHERE Goods LIKE 'Вафли%'

9.	Выведите среднюю цену на растительное масло (закупочную).

SELECT AVG(CostUnit) AS 'Средняя цена'
FROM SallerGoods
WHERE Goods='масло подсолнечное'

10.	Определите сколько всего было продано килограмм яблок.

SELECT SUM(cg.Count) AS 'сколько всего было продано килограмм яблок'
FROM ClientGoods cg, SallerGoods as sg
WHERE cg.ID=sg.ID AND sg.Goods LIKE 'Яблоки%'

11.	Определите какое количество картофеля закупили кафе.

SELECT SUM(cg.Count) AS 'какое количество картофеля закупили кафе, кг.', cg.ID
FROM ClientGoods cg, SallerGoods as sg, Client c
WHERE cg.INNClient=c.INNClient AND cg.ID = sg.ID AND c.Status LIKE 'кафе' AND sg.Goods LIKE 'Картофель%';	

12.	Посчитать сколько клиентов купили перец черный молотый.

SELECT COUNT(c.Client) AS 'сколько клиентов купили перец черный молотый'
FROM ClientGoods cg, SallerGoods as sg, Client c
WHERE cg.INNClient=c.INNClient AND cg.ID = sg.ID AND sg.Goods LIKE 'Перец черный молотый';

13.	Определить сколько наименований товара было куплено у посредников.

SELECT COUNT(DISTINCT sg.Goods) AS 'сколько наименований товара было куплено у посредников'
FROM ClientGoods cg, SallerGoods sg, Seller s
WHERE cg.ID=sg.ID AND sg.INNSeller=s.INNSeller AND s.Status LIKE 'посредник'

14.	Определить минимальную цену за единицу проданного товара.

SELECT MIN(CostUnit) AS 'минимальная цена за единицу проданного товара'
FROM ClientGoods

15.	Определите минимальную сумму на какую было закуплено товара.

SELECT MIN(CostUnit*Count) AS 'минимальная сумма на какую было закуплено товара'
FROM SallerGoods

16.	Определить максимальную сумму за товар, проданный магазинам.

SELECT MAX(CostUnit*Count) AS 'максимальная сумма за товар, проданный магазинам'
FROM ClientGoods cg, Client c
WHERE cg.INNClient = c.INNClient AND c.Status LIKE 'магазин'

17.	Определить со сколькими фирмами сотрудничает данная фирма.

SELECT COUNT(DISTINCT c.Client) AS 'Количество фирм'
FROM SallerGoods sg, Client c, ClientGoods cg, Seller s
WHERE c.INNClient=cg.INNClient AND cg.ID=sg.ID AND sg.INNSeller=s.INNSeller AND s.INNSeller LIKE '1023'

18.	Определить минимальную сумму сделки по закупке товара у производителей.

SELECT MIN(cg.CostUnit*cg.Count) AS 'минимальная сумма сделки по закупке товара у производителей'
FROM Client c, ClientGoods cg, SallerGoods sg, Seller s
WHERE c.INNClient=cg.INNClient AND cg.ID=sg.ID AND sg.INNSeller=s.INNSeller AND s.Status LIKE 'производитель'


19.	Определить среднюю цену за пачку чая при покупке.

SELECT AVG(cg.CostUnit) AS 'Средняя цена за пачку чая при покупке'
FROM ClientGoods cg, SallerGoods sg
WHERE cg.ID=sg.ID AND sg.Goods LIKE 'Чай%'

20.	Определите максимальную сумму прибыли за товар, проданный посреднику.

SELECT MAX(cg.CostUnit*cg.Count)-(cg.Count*sg.CostUnit) AS 'Максимальная прибыль'
FROM Client c, ClientGoods cg, SallerGoods sg
WHERE c.INNClient=cg.INNClient AND Status LIKE '%посредник%'

21.	 Определите минимальную прибыль за товар, купленный у посредника.

SELECT MIN(cg.CostUnit*cg.Count)-(cg.Count*sg.CostUnit) AS 'Минимальная прибыль'
FROM Client c, ClientGoods cg, SallerGoods sg, Seller s
WHERE c.INNClient=cg.INNClient AND cg.ID=sg.ID AND sg.INNSeller=s.INNSeller AND s.Status LIKE 'посредник'

22.	 Какое количество наименований товара куплено у фирмы.

SELECT COUNT(DISTINCT cg.ID) AS 'Кол-во наименований'
FROM SallerGoods sg, ClientGoods as cg, Seller s
WHERE sg.ID = cg.ID AND sg.INNSeller = s.INNSeller AND s.Seller= '"Океан"'

23.	 Сколько в продаже видов печенья.

SELECT COUNT(DISTINCT Goods) AS 'Сколько в продаже видов печенья'
FROM SallerGoods
WHERE Goods LIKE 'Печенье%'

24.	 Какая максимальная сумма покупки, на которую клиент сделал заказ.

SELECT MAX(CostUnit*Count) AS 'максимальная сумма покупки'
FROM ClientGoods

25.	 Определите среднюю прибыль от перепродажи масла.

SELECT AVG((cg.CostUnit - sg.CostUnit) * cg.Count) AS 'среднюю прибыль от перепродажи масла'
FROM SallerGoods sg, ClientGoods as cg
WHERE sg.ID = cg.ID AND sg.Goods LIKE 'масло%'

26.	 Со сколькими посредниками сотрудничает фирма.

SELECT COUNT(DISTINCT c.Client) AS 'Количество посредников'
FROM SallerGoods sg, Client c, ClientGoods cg, Seller s
WHERE c.INNClient=cg.INNClient AND cg.ID=sg.ID AND sg.INNSeller=s.INNSeller AND c.Status LIKE 'посредник'
